```typescript
// apps/desktop/src/main/handlers/git.ts
import simpleGit, { type SimpleGit } from 'simple-git'
import type { GitStatus, GitBranch, GitLogEntry } from './git.schema'

export class GitHandler {
  private git(repoPath: string): SimpleGit {
    return simpleGit(repoPath)
  }

  async status(repoPath: string): Promise<GitStatus> {
    const git = this.git(repoPath)
    const result = await git.status()

    return {
      current: result.current,
      tracking: result.tracking,
      staged: result.staged,
      unstaged: result.modified,
      untracked: result.not_added,
      conflicted: result.conflicted,
    }
  }

  async branches(repoPath: string): Promise<GitBranch[]> {
    const git = this.git(repoPath)
    const branchSummary = await git.branch()

    return branchSummary.all.map((name) => ({
      name,
      current: name === branchSummary.current,
    }))
  }

  async currentBranch(repoPath: string): Promise<string> {
    const git = this.git(repoPath)
    const branchSummary = await git.branch()
    return branchSummary.current
  }

  async log(repoPath: string, count = 50): Promise<GitLogEntry[]> {
    const git = this.git(repoPath)
    const result = await git.log({ maxCount: count, '--no-optional-locks': '' })

    return result.all.map((entry) => ({
      hash: entry.hash,
      date: entry.date,
      message: entry.message,
      author_name: entry.author_name,
      author_email: entry.author_email,
    }))
  }

  async diffStat(repoPath: string): Promise<{ additions: number; deletions: number }> {
    const git = this.git(repoPath)
    const result = await git.diff(['--stat'])

    // Parse diff --stat output
    const lines = result.split('\n')
    let additions = 0
    let deletions = 0

    for (const line of lines) {
      const match = line.match(/\+(\d+).*-(\d+)/)
      if (match) {
        additions += parseInt(match[1], 10)
        deletions += parseInt(match[2], 10)
      }
    }

    return { additions, deletions }
  }

  async stage(repoPath: string, files: string[]): Promise<void> {
    const git = this.git(repoPath)
    await git.add(files)
  }

  async unstage(repoPath: string, files: string[]): Promise<void> {
    const git = this.git(repoPath)
    await git.reset(['HEAD', '--', ...files])
  }

  async stageAll(repoPath: string): Promise<void> {
    const git = this.git(repoPath)
    await git.add('-A')
  }

  async unstageAll(repoPath: string): Promise<void> {
    const git = this.git(repoPath)
    await git.reset(['HEAD'])
  }

  async discard(repoPath: string, files: string[]): Promise<void> {
    const git = this.git(repoPath)
    await git.checkout(['--', ...files])
  }

  async commit(repoPath: string, message: string): Promise<{ hash: string }> {
    const git = this.git(repoPath)
    const result = await git.commit(message, ['--no-optional-locks'])
    return { hash: result.commit }
  }

  async checkout(repoPath: string, branch: string): Promise<{ success: boolean }> {
    const git = this.git(repoPath)
    try {
      await git.checkout(branch)
      return { success: true }
    } catch {
      return { success: false }
    }
  }

  async createBranch(repoPath: string, name: string): Promise<void> {
    const git = this.git(repoPath)
    await git.checkoutLocalBranch(name)
  }

  async push(repoPath: string): Promise<{ success: boolean; message?: string }> {
    const git = this.git(repoPath)
    try {
      const result = await git.push(['--no-optional-locks'])
      return { success: true, message: result.pushed.join(', ') }
    } catch (err) {
      return { success: false, message: String(err) }
    }
  }

  async pull(repoPath: string): Promise<{ success: boolean; message?: string }> {
    const git = this.git(repoPath)
    try {
      await git.pull(['--no-optional-locks'])
      return { success: true }
    } catch (err) {
      return { success: false, message: String(err) }
    }
  }

  async fetch(repoPath: string): Promise<void> {
    const git = this.git(repoPath)
    await git.fetch(['--all', '--no-optional-locks'])
  }

  // TODO: generateCommitMessage - requires AI integration
  async generateCommitMessage(repoPath: string): Promise<string> {
    // Placeholder - will be implemented with AI integration
    const status = await this.status(repoPath)
    const changes = status.staged.length + status.unstaged.length + status.untracked.length
    return `Update ${changes} files`
  }

  // Registration

  }
}
```

```typescript
// apps/desktop/src/renderer/src/features/chat/hooks/useGit.ts
import { useAtom } from 'jotai'
import { useCallback, useEffect, useRef } from 'react'

import { gitApi } from '@renderer/api'
import { currentProjectAtom } from '@renderer/stores'

interface UseGitStatusResult {
  status: GitStatus | null
  branches: GitBranch[]
  currentBranch: string
  log: GitLogEntry[]
  diffStat: { additions: number; deletions: number }
  loading: boolean
  error: string | null
  refresh: () => void
}

export function useGitStatus(pollIntervalMs = 3000): UseGitStatusResult {
  const [project] = useAtom(currentProjectAtom)
  const [status, setStatus] = useAtom(gitStatusAtom)
  const [branches, setBranches] = useAtom(gitBranchesAtom)
  const [currentBranch, setCurrentBranch] = useAtom(gitCurrentBranchAtom)
  const [log, setLog] = useAtom(gitLogAtom)
  const [diffStat, setDiffStat] = useAtom(gitDiffStatAtom)
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState<string | null>(null)

  const fetchIdRef = useRef(0)
  const intervalRef = useRef<ReturnType<typeof setInterval>>()

  const fetchGitData = useCallback(async () => {
    if (!project?.path) return

    const id = ++fetchIdRef.current
    setLoading(true)
    setError(null)

    try {
      const [statusResult, branchesResult, branchResult, logResult, diffResult] =
        await Promise.all([
          gitApi.status(project.path),
          gitApi.branches(project.path),
          gitApi.currentBranch(project.path),
          gitApi.log(project.path, 50),
          gitApi.diffStat(project.path),
        ])

      if (id !== fetchIdRef.current) return

      setStatus(statusResult)
      setBranches(branchesResult)
      setCurrentBranch(branchResult)
      setLog(logResult)
      setDiffStat(diffResult)
    } catch (err) {
      if (id !== fetchIdRef.current) return
      setError(err instanceof Error ? err.message : 'Failed to fetch git data')
    } finally {
      if (id === fetchIdRef.current) {
        setLoading(false)
      }
    }
  }, [project?.path])

  // Polling with visibility check
  useEffect(() => {
    if (!project?.path) return

    fetchGitData()

    const startPolling = () => {
      intervalRef.current = setInterval(() => {
        if (document.visibilityState === 'visible') {
          fetchGitData()
        }
      }, pollIntervalMs)
    }

    startPolling()

    const handleVisibilityChange = () => {
      if (document.visibilityState === 'visible') {
        fetchGitData()
      }
    }

    document.addEventListener('visibilitychange', handleVisibilityChange)

    return () => {
      clearInterval(intervalRef.current)
      document.removeEventListener('visibilitychange', handleVisibilityChange)
    }
  }, [project?.path, pollIntervalMs, fetchGitData])

  return {
    status,
    branches,
    currentBranch,
    log,
    diffStat,
    loading,
    error,
    refresh: fetchGitData,
  }
}
```

```tsx
// apps/desktop/src/renderer/src/features/chat/components/panel/git/ChangesSection.tsx
import { useMemo, useCallback } from 'react'
import { useGitStatus } from '../../hooks/useGit'
import { useGitActions } from '../../hooks/useGitActions'
import { HugeiconsFile } from '@hugeicons/react'

interface ChangesSectionProps {}

export function ChangesSection({}: ChangesSectionProps) {
  const { status, loading, error } = useGitStatus()
  const { stage, unstage, stageAll, discard } = useGitActions()

  const stagedFiles = useMemo(() => status?.staged || [], [status])
  const unstagedFiles = useMemo(() => [...(status?.unstaged || []), ...(status?.untracked || [])], [status])

  if (loading && !status) {
    return <div className="p-4 text-xs text-muted-foreground">Loading...</div>
  }

  if (error) {
    return <div className="p-4 text-xs text-destructive">{error}</div>
  }

  return (
    <div className="flex flex-col gap-4 p-3">
      {/* Staged changes */}
      <div className="flex flex-col gap-1">
        <div className="flex items-center justify-between">
          <span className="text-[10px] font-semibold text-muted-foreground/70">
            Staged ({stagedFiles.length})
          </span>
          {stagedFiles.length > 0 && (
            <button
              onClick={() => unstage(stagedFiles)}
              className="text-[10px] text-muted-foreground hover:text-foreground"
            >
              Unstage All
            </button>
          )}
        </div>
        {stagedFiles.map((file) => (
          <div
            key={file}
            className="flex items-center gap-2 rounded px-2 py-1 hover:bg-foreground/[0.05]"
          >
            <HugeiconsFile className="h-3 w-3 text-green-500" />
            <span className="text-xs truncate">{file}</span>
          </div>
        ))}
      </div>

      {/* Unstaged changes */}
      <div className="flex flex-col gap-1">
        <div className="flex items-center justify-between">
          <span className="text-[10px] font-semibold text-muted-foreground/70">
            Changes ({unstagedFiles.length})
          </span>
          {unstagedFiles.length > 0 && (
            <button
              onClick={() => stageAll()}
              className="text-[10px] text-muted-foreground hover:text-foreground"
            >
              Stage All
            </button>
          )}
        </div>
        {unstagedFiles.map((file) => (
          <div
            key={file}
            className="flex items-center gap-2 rounded px-2 py-1 hover:bg-foreground/[0.05]"
          >
            <HugeiconsFile className="h-3 w-3 text-orange-500" />
```

```tsx
// apps/desktop/src/renderer/src/features/chat/components/panel/git/CommitsSection.tsx
import { useMemo } from 'react'
import { useGitStatus } from '../../hooks/useGit'
import { HugeiconsTime } from '@hugeicons/react'

interface CommitsSectionProps {}

export function CommitsSection({}: CommitsSectionProps) {
  const { log, loading, error } = useGitStatus()

  if (loading && log.length === 0) {
    return <div className="p-4 text-xs text-muted-foreground">Loading...</div>
  }

  if (error) {
    return <div className="p-4 text-xs text-destructive">{error}</div>
  }

  if (log.length === 0) {
    return <div className="p-4 text-xs text-muted-foreground">No commits yet</div>
  }

  return (
    <div className="flex flex-col gap-1 p-3">
      {log.map((entry) => (
        <div
          key={entry.hash}
          className="flex flex-col gap-0.5 rounded px-2 py-2 hover:bg-foreground/[0.05]"
        >
          <div className="flex items-center gap-2">
            <HugeiconsTime className="h-3 w-3 shrink-0 text-muted-foreground/50" />
            <span className="text-xs font-medium truncate">{entry.message}</span>
          </div>
          <div className="flex items-center gap-2 text-[10px] text-muted-foreground/60">
            <span className="font-mono">{entry.hash.slice(0, 7)}</span>
            <span>{entry.author_name}</span>
            <span>{new Date(entry.date).toLocaleDateString()}</span>
          </div>
        </div>
      ))}
    </div>
  )
}
```
